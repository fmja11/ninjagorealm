PROFILES (extends built-in auth.users with app data) create table if not exists profiles ( id uuid primary key references auth.users(id) on delete cascade, name text not null, email text not null, role text not null default 'buyer' check (role in ('buyer','seller','admin','owner')), city text, banned boolean not null default false, rating numeric not null default 0, rating_count integer not null default 0, avatar_seed text, created_at timestamptz not null default now() ); -- Auto-create a profile row whenever someone signs up create or replace function handle_new_user() returns trigger as $$ begin insert into profiles (id, name, email, role, city, avatar_seed) values ( new.id, coalesce(new.raw_user_meta_data->>'name', 'New user'), new.email, coalesce(new.raw_user_meta_data->>'role', 'buyer'), new.raw_user_meta_data->>'city', upper(left(coalesce(new.raw_user_meta_data->>'name','N'),1)) ); return new; end; $$ language plpgsql security definer; drop trigger if exists on_auth_user_created on auth.users; create trigger on_auth_user_created after insert on auth.users for each row execute procedure handle_new_user(); -- 2. LISTINGS create table if not exists listings ( id uuid primary key default gen_random_uuid(), seller_id uuid references profiles(id) on delete cascade, title text not null, set_number text, category text not null, condition text, city text, price integer, original_price integer, year integer, pieces integer, description text, icon text default '🥷', status text not null default 'pending' check (status in ('pending','active','sold','rejected','removed')), featured boolean not null default false, views integer not null default 0, created_at timestamptz not null default now() ); -- 3. ORDERS create table if not exists orders ( id uuid primary key default gen_random_uuid(), listing_id uuid references listings(id), listing_title text, buyer_id uuid references profiles(id), seller_id uuid references profiles(id), amount integer not null, carrier text, status text not null default 'processing' check (status in ('processing','shipped','delivered','cancelled','disputed')), created_at timestamptz not null default now() ); -- 4. OFFERS create table if not exists offers ( id uuid primary key default gen_random_uuid(), listing_id uuid references listings(id), buyer_id uuid references profiles(id), seller_id uuid references profiles(id), amount integer not null, message text, status text not null default 'pending' check (status in ('pending','accepted','declined')), created_at timestamptz not null default now() ); -- 5. REVIEWS create table if not exists reviews ( id uuid primary key default gen_random_uuid(), listing_id uuid references listings(id), author_id uuid references profiles(id), author_name text, stars integer not null check (stars between 1 and 5), text text, created_at timestamptz not null default now() ); -- 6. MESSAGES create table if not exists messages ( id uuid primary key default gen_random_uuid(), sender_id uuid references profiles(id), recipient_id uuid references profiles(id), listing_id uuid references listings(id), body text not null, read boolean not null default false, created_at timestamptz not null default now() ); -- 7. WISHLISTS create table if not exists wishlists ( user_id uuid references profiles(id) on delete cascade, listing_id uuid references listings(id) on delete cascade, created_at timestamptz not null default now(), primary key (user_id, listing_id) ); -- 8. AUDIT LOG create table if not exists audit_log ( id uuid primary key default gen_random_uuid(), actor_id uuid references profiles(id), action text not null, detail text, created_at timestamptz not null default now() ); -- 9. SITE META (single-row config: fees, site name) create table if not exists site_meta ( id integer primary key default 1, site_name text not null default 'Ninjago Realm', platform_fee_rate numeric not null default 0.06, owner_email text, constraint single_row check (id = 1) ); insert into site_meta (id, site_name, platform_fee_rate, owner_email) values (1, 'Ninjago Realm', 0.06, null) on conflict (id) do nothing; -- ============================================================ -- ROW LEVEL SECURITY — every table locked down by default, -- then opened up only for the exact access each role needs. -- ============================================================ alter table profiles enable row level security; alter table listings enable row level security; alter table orders enable row level security; alter table offers enable row level security; alter table reviews enable row level security; alter table messages enable row level security; alter table wishlists enable row level security; alter table audit_log enable row level security; alter table site_meta enable row level security; -- helper: is the current user an admin or owner? create or replace function is_staff() returns boolean as $$ select exists ( select 1 from profiles where id = auth.uid() and role in ('admin','owner') ); $$ language sql security definer stable; -- PROFILES: anyone can read (needed for seller names/ratings), only the -- owner of a row (or staff) can update it. create policy "profiles_select_all" on profiles for select using (true); create policy "profiles_update_own" on profiles for update using (auth.uid() = id or is_staff()); -- LISTINGS: active listings are public; sellers see their own of any status; -- staff see everything. Sellers insert/update their own; staff can update any. create policy "listings_select" on listings for select using ( status = 'active' or seller_id = auth.uid() or is_staff() ); create policy "listings_insert_own" on listings for insert with check (seller_id = auth.uid()); create policy "listings_update_own_or_staff" on listings for update using ( seller_id = auth.uid() or is_staff() ); create policy "listings_delete_staff" on listings for delete using (is_staff()); -- ORDERS: only buyer, seller, or staff can see an order create policy "orders_select" on orders for select using ( buyer_id = auth.uid() or seller_id = auth.uid() or is_staff() ); create policy "orders_insert_buyer" on orders for insert with check (buyer_id = auth.uid()); create policy "orders_update" on orders for update using ( buyer_id = auth.uid() or seller_id = auth.uid() or is_staff() ); -- OFFERS: only buyer, seller, or staff can see an offer create policy "offers_select" on offers for select using ( buyer_id = auth.uid() or seller_id = auth.uid() or is_staff() ); create policy "offers_insert_buyer" on offers for insert with check (buyer_id = auth.uid()); create policy "offers_update_seller" on offers for update using ( seller_id = auth.uid() or is_staff() ); -- REVIEWS: public read, only the author can write their own create policy "reviews_select_all" on reviews for select using (true); create policy "reviews_insert_own" on reviews for insert with check (author_id = auth.uid()); -- MESSAGES: only sender or recipient can read; only sender can insert as themself create policy "messages_select" on messages for select using ( sender_id = auth.uid() or recipient_id = auth.uid() or is_staff() ); create policy "messages_insert_own" on messages for insert with check (sender_id = auth.uid()); create policy "messages_update_own" on messages for update using ( recipient_id = auth.uid() ); -- WISHLISTS: only the owner of the wishlist can see/change it create policy "wishlists_own" on wishlists for all using (user_id = auth.uid()) with check (user_id = auth.uid()); -- AUDIT LOG: staff only create policy "audit_select_staff" on audit_log for select using (is_staff()); create policy "audit_insert_any" on audit_log for insert with check (auth.uid() is not null); -- SITE META: everyone can read, only staff can change create policy "meta_select_all" on site_meta for select using (true); create policy "meta_update_staff" on site_meta for update using (is_staff());
